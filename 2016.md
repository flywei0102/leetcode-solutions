class Solution {
    public int maximumDifference(int[] nums) {
        int max = 0;
        int min = Integer.MAX_VALUE;
        for(int i = 0; i < nums.length; i++){
            if(nums[i] < min){
                min = nums[i];
            }else if(max < nums[i] - min){
                max = nums[i] - min;
            }
        }
        if(max == 0) return -1;
        return max;
    }
}
