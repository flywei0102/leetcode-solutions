class Solution {

​    public int search(int[] nums, int target) {

​        if(target < nums[0] || target > nums[nums.length - 1]){

​            return -1;

​        }

​        

​        int left = 0;

​        int right = nums.length - 1;

​        

​        while(left <= right){

​            int mid = left + (int)Math.floor((right - left)/2);

​            if(nums[mid] == target){

​                return mid;

​            }else if(nums[mid] < target){

​                left = mid + 1;

​            }else if(nums[mid] > target){

​                right = mid - 1;

​            }

​        }

​        return -1;

​    }

}
